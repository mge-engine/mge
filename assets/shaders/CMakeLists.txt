SET(asset_files
    triangle.frag
    triangle.vert
    triangle.frag.spv
    triangle.vert.spv
)

FOREACH(ASSET_FILE ${asset_files})
    MESSAGE(STATUS "Copying asset file ${ASSET_FILE} to ${CMAKE_CURRENT_BINARY_DIR}")

ENDFOREACH(ASSET_FILE)
